Result

Once you've got the result you should always check if an error has occurred, if not, retrieve the resource.

use Railken\Amethyst\Managers\PostManager;

$manager = new PostManager();

$result = $manager->create([
    "title" => "Maia Gleichner",
    "body" => "Temporibus consectetur voluptatem odio aliquid. Eos aspernatur et tempore dolores consectetur similique laudantium. Pariatur voluptas consequatur et sint.",
    "postable_type" => "foo",
    "postable" => [
        "name" => "Minerva McDermott",
        "description" => "Recusandae nobis unde adipisci iure voluptatem. Quod qui dignissimos consequatur maiores vel qui. Qui accusantium dolorem necessitatibus.",
        "bar" => [
            "name" => "Miss Bonnie DuBuque MD",
            "description" => "Et fuga quia dolorum quas velit et ut consectetur. Eveniet et quia sunt sed aut ratione eum. Ipsam aspernatur voluptas ex natus vel fugit. Repudiandae nesciunt ipsa eius est aut."
        ]
    ]
]);

if ($result->ok()) {

    $resource = $result->getResource();

} else {

    // Loop through all errors
    $result->getErrors()->map(function($error) {
        return $error->toArray();
    }))

    // Retrieve an array of all errors
    $result->getSimpleErrors();

    /* The result is something like this:

        [0] => Array
            (
                [code] => FIELD_NOT_DEFINED
                [label] => field
                [message] => The field is required
                [value] =>
            )
    */

}
